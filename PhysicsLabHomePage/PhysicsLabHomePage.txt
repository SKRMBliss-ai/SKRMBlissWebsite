      sel.addEventListener('change', (e) => { localStorage.setItem('skrm_lang', e.target.value); });
    })();

    (function(){
      const btn = document.getElementById('sim-call-btn');
      const phone = '+919902236432';
      let tTimer = null;
      function showToast(text){
        const existing = document.querySelector('.sim-toast');
        if(existing) existing.remove();
        const t = document.createElement('div');
        t.className = 'sim-toast';
        t.textContent = text + ' (copied)';
               t.style.cssText = 'position:fixed;bottom:90px;right:20px;background:#071026;color:white;padding:12px 20px;border-radius:12px;font-weight:700;box-shadow:0 12px 36px rgba(2,6,23,0.18);z-index:99999;opacity:0;transform:translateY(8px);transition:all .32s ease';
        document.body.appendChild(t);
        // allow transition
        requestAnimationFrame(() => { t.style.opacity = '1'; t.style.transform = 'translateY(0)'; });

        clearTimeout(tTimer);
        tTimer = setTimeout(() => {
          t.style.opacity = '0';
          t.style.transform = 'translateY(8px)';
          setTimeout(() => t.remove(), 350);
        }, 2600);
      }

      if(btn){
        btn.addEventListener('click', async (e) => {
          e.preventDefault();
          try{
            if(navigator.clipboard && navigator.clipboard.writeText){
              await navigator.clipboard.writeText(phone);
              showToast(phone);
            } else {
              // fallback
              const ta = document.createElement('textarea');
              ta.value = phone;
              ta.style.position = 'fixed';
              ta.style.left = '-9999px';
              document.body.appendChild(ta);
              ta.select();
              document.execCommand('copy');
              ta.remove();
              showToast(phone);
            }
            // also temporarily change button label for visual feedback
            const prev = btn.innerHTML;
            btn.innerHTML = phone;
            setTimeout(()=> btn.innerHTML = prev, 2500);
          }catch(err){
            console.warn('copy failed', err);
            showToast('Unable to copy, please dial: ' + phone);
          }
        });

        // keyboard support
        btn.addEventListener('keydown', (ev) => {
          if(ev.key === 'Enter' || ev.key === ' '){
            ev.preventDefault();
            btn.click();
          }
        });
      }

      // Make simulator previews act as accessible launch targets:
      // clicking the preview finds the "Launch Simulator" button in the same card and triggers it.
      document.querySelectorAll('.sim-preview').forEach(preview => {
        preview.setAttribute('role','button');
        preview.setAttribute('tabindex','0');
        preview.setAttribute('aria-label','Open simulator');
        preview.addEventListener('click', (ev) => {
          ev.preventDefault();
          const card = preview.closest('.sim-card');
          if(!card) return;
          const launch = card.querySelector('.sim-cta .btn');
          if(launch) launch.click();
        });
        preview.addEventListener('keydown', (ev) => {
          if(ev.key === 'Enter' || ev.key === ' '){
            ev.preventDefault();
            preview.click();
          }
        });
      });

      // Close mobile menu on navigation link click (small screens)
      document.querySelectorAll('#sim-mobile-menu a').forEach(a => {
        a.addEventListener('click', () => {
          mm.style.display = 'none';
          hb.setAttribute('aria-expanded','false');
        });
      });

      // Small helper: ensure external links open safely
      document.querySelectorAll('a[target="_blank"]').forEach(a => {
        a.setAttribute('rel', (a.getAttribute('rel') || '') + ' noopener noreferrer');
      });
    })();
  })();
